#### 상속 / 오버라이딩

> 상속이란, <br>
> super class(부모) 및 sub class(자식) -> 모든 속성, 메서드 사용가능 <br>

```python
# 부모 클래스
class Car:
    """
    class 밑에 이렇게 주석을 달아서 그 클래스에 대한 설명을 적어줘야한다.
    그것이 PEP8 원칙이다.
    - Parent Class -
    """
    
    def __init__(self, tp, color):
	self.type = tp
	self.color = color
    
    def show(self):
	return "Car Class 'show method'"

# 자식 클래스
class BmwCar(Car):
    """
    Sub Class
    """

    def __init__(self, car_name, tp, color):
        super().__init__(tp, color)
        self.car_name = car_name

    def show_model(self):
        return "Your Car Name : '%s'" % self.car_name

# 자식 클래스
class BenzCar(Car):
    def __init__(self, car_name, tp, color):
        super().__init__(tp, color)
        self.car_name = car_name

    def show_model(self):
        return "Your Car Name : '%s'" % self.car_name

    def show(self):  # 부모한테도 있는 메서드
        print(super().show()) # 부모 클래스에 있는걸 쓰고 싶으면 이렇게 쓰면된다.
        # 이름만 똑같고 코드 내용만 다르다.
        return "Car Info :  %s %s %s " % (self.car_name, self.type, self.color)

# 일반 사용방법
model1 = BmwCar("520d", "sedan", "red")
print(model1.color)  # super
print(model1.type)  # super
print(model1.car_name)  # sub
print(model1.show())  # super
print(model1.show_model())  # sub
print(model1.__dict__)

# Method Overriding(오버라이딩) - > 올라타다
model2 = BenzCar("220d", "suv", "black")
print(model2.show())
# 자식에서 상속 받을건 받고 기능을 개선하거나 추가하거나 입맛대로 개선할때
# 코드 재활용 가능

# Parent Method Call
model3 = BenzCar("350s", "sedan", "silver")
print(model3.show())

# Inheritance Info (상속 정보)
print(BmwCar.mro())  # 상속받은 순서를 보여준다.
print(BenzCar.mro())

```

#### 다중 상속

> 다중 상속은 간단하게 쓰면 보기 쉽고 유용하게 쓸 수 있지만, 다중 상속을 많이 쓰게 되면 아직 갓 입문한 주니어 개발자에겐 코드의 복잡도가 상승되고 코드가 꼬일 가능성이 높다.

```python
# 다중 상속의 예제
class X():
    pass


class Y():
    pass


class Z():
    pass


class A(X, Y):
    pass


class B(Y, Z):
    pass


class M(B, A, Z):
    pass


print(M.mro()) # 상속의 순서를 알아 볼 수 있다.
```
